## 3. Searching & Checking

| Method                      | Description                            | Returns                  |

```js
"hello".startsWith("he"); // true

// matchAll (ES2020) - Returns iterator of all matches

const str = "test1test2";
const matches = str.matchAll(regex);
for (const match of matches) {
  console.log(match[0]); // "test1", "test2"
  console.log(match[1]); // "e", "e"
  console.log(match[2]); // "st1", "st2"
}
```

## 8. Useful Concepts

- Strings are immutable: You cannot change a character directly
- Use + or .concat() to combine strings
- Beware of Unicode (emoji, surrogate pairs) which can affect length and indexing

## Common Use Cases

### 1. Input Validation & Sanitization

```js
// Trim user input
const userInput = "  hello@example.com  ".trim();

// Validate email format
function isValidEmail(email) {
  return email.includes("@") && email.includes(".");
}

// Check password strength
function hasMinLength(str, min) {
  return str.length >= min;
}
```

### 2. URL & Path Manipulation

```js
// Extract domain
const url = "https://example.com/path";
const domain = url.split("//")[1].split("/")[0]; // "example.com"

// Build query string
const params = { name: "John", age: 30 };
const queryString = Object.entries(params)
  .map(([k, v]) => `${encodeURIComponent(k)}=${encodeURIComponent(v)}`)
  .join("&"); // "name=John&age=30"
```

### 3. Text Processing

```js
// Capitalize first letter
function capitalize(str) {
  return str.charAt(0).toUpperCase() + str.slice(1).toLowerCase();
}

// Reverse string
function reverse(str) {
  return str.split("").reverse().join("");
}

// Count words
function countWords(str) {
  return str.trim().split(/\s+/).length;
}
```

### 4. Formatting

```js
// Format phone number
function formatPhone(phone) {
  return phone.replace(/(\d{3})(\d{3})(\d{4})/, "($1) $2-$3");
}

// Format currency
function formatCurrency(amount) {
  return "$" + amount.toFixed(2).replace(/\B(?=(\d{3})+(?!\d))/g, ",");
}
```

## Interview Common Questions

### Q: Check if string is a palindrome?

```js
function isPalindrome(str) {
  const cleaned = str.toLowerCase().replace(/[^a-z0-9]/g, "");
  return cleaned === cleaned.split("").reverse().join("");
}

isPalindrome("A man a plan a canal Panama"); // true
```

### Q: Find longest word in string?

```js
function longestWord(str) {
  return str
    .split(/\s+/)
    .reduce((longest, word) => (word.length > longest.length ? word : longest));
}
```

### Q: Check if two strings are anagrams?

```js
function areAnagrams(str1, str2) {
  const normalize = (str) => str.toLowerCase().split("").sort().join("");
  return normalize(str1) === normalize(str2);
}

areAnagrams("listen", "silent"); // true
```

### Q: Difference between slice, substring, and substr?

- **`slice(start, end)`**: Can use negative indices, doesn't swap if start > end
- **`substring(start, end)`**: Swaps if start > end, treats negatives as 0
- **`substr(start, length)`**: Deprecated, use `slice()` instead

```js
"hello".slice(1, 4); // "ell" (from index 1 to 3)
"hello".slice(-3); // "llo" (last 3 chars, negative works)
"hello".substring(4, 1); // "ell" (swaps to 1, 4)
"hello".substring(-3); // "hello" (negative becomes 0)
```

### Q: How to reverse a string?

```js
// Method 1: Split, reverse, join
"hello".split("").reverse().join(""); // "olleh"

// Method 2: Loop backwards
function reverse(str) {
  let result = "";
  for (let i = str.length - 1; i >= 0; i--) {
    result += str[i];
  }
  return result;
}
```

### Q: Check if string contains substring?

```js
// Modern way (ES6+)
str.includes("substring"); // Returns boolean

// Old way
str.indexOf("substring") !== -1; // Returns boolean
```

### Q: Remove duplicates from string?

```js
function removeDuplicates(str) {
  return [...new Set(str)].join("");
}

removeDuplicates("hello"); // "helo"
```
