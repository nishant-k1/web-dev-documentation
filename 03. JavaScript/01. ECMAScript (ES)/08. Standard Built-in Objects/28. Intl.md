### `Intl.DateTimeFormat` ‚≠ê‚≠ê‚≠ê‚≠ê‚≠ê

Format dates and times.

```js
const date = new Date("2023-12-25");

// US format
new Intl.DateTimeFormat("en-US").format(date); // "12/25/2023"

// UK format
new Intl.DateTimeFormat("en-GB").format(date); // "25/12/2023"

// With options
new Intl.DateTimeFormat("en-US", {
  year: "numeric",
  month: "long",
  day: "numeric",
}).format(date); // "December 25, 2023"
```

### `Intl.NumberFormat` ‚≠ê‚≠ê‚≠ê‚≠ê‚≠ê

Format numbers, currencies, and percentages.

```js
// Currency formatting

  style: "currency",
  currency: "USD",
}).format(1234.56); // "$1,234.56"

// Percentage

  style: "percent",
}).format(0.123); // "12%"

// Locale-specific number format
new Intl.NumberFormat("de-DE").format(1234.56); // "1.234,56" (German format)
```

```js

pr.select(2); // "other"
```

### `Intl.RelativeTimeFormat` ‚≠ê‚≠ê‚≠ê

Format relative time.

```js

rtf.format(-1, "day"); // "yesterday"
rtf.format(1, "hour"); // "in 1 hour"
rtf.format(2, "month"); // "in 2 months"
```

### `Intl.ListFormat` ‚≠ê‚≠ê

Format lists.


### `Intl.Locale` ‚≠ê‚≠ê

Represents a locale identifier.

## Common Use Cases

### 1. Date Formatting


### 4. Relative Time

```js
const rtf = new Intl.RelativeTimeFormat("en");

function formatRelativeTime(date) {
  const seconds = Math.floor((date - Date.now()) / 1000);
  if (Math.abs(seconds) < 60) return rtf.format(seconds, "second");
  if (Math.abs(seconds) < 3600)
    return rtf.format(Math.floor(seconds / 60), "minute");
  if (Math.abs(seconds) < 86400)
    return rtf.format(Math.floor(seconds / 3600), "hour");
  return rtf.format(Math.floor(seconds / 86400), "day");
}
```
