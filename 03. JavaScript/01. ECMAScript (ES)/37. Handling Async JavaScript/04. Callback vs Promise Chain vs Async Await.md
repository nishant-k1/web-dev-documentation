# Callback vs Promise Chain vs Async Await

An example of callback hell and how same is handled with promise chain and async await

If each async operation is independent and doesn't depend on the result of the previous one, you don't even need to chain them sequentially. You can run them in parallel using Promise.all(), which is even cleaner. Nesting only becomes necessary when one operation depends on the result of a previous one. If they're independent, you can handle them separately or in parallel without any nesting at all.

## Callback

```js
getUser(userId, function (user) {
  getPosts(user.id, function (posts) {
    getComments(posts[0].id, function (comments) {
      getLikes(comments[0].id, function (likes) {
        console.log(likes);
        // keep going deeper and deeper...
      });
    });
  });
});
```

## promise chain

```js
getUser(userId)
  .then(function (user) {
    return getPosts(user.id);
  })
  .then(function (posts) {
    return getComments(posts[0].id);
  })
  .then(function (comments) {
    return getLikes(comments[0].id);
  })
  .then(function (likes) {
    console.log(likes);
  })
  .catch(function (error) {
    console.log(error);
  });
```

## async await

```js
async function getData() {
  const user = await getUser(userId);
  const posts = await getPosts(user.id);
  const comments = await getComments(posts[0].id);
  const likes = await getLikes(comments[0].id);
  console.log(likes);
}
```

## Real complete implementation using callback

```js
function getUser(userId, callback) {
  setTimeout(function () {
    const user = { id: userId, name: "Nishant" };
    callback(user);
  }, 1000);
}

function getPosts(userId, callback) {
  setTimeout(function () {
    const posts = [
      { id: 1, title: "Post 1" },
      { id: 2, title: "Post 2" },
    ];
    callback(posts);
  }, 1000);
}

function getComments(postId, callback) {
  setTimeout(function () {
    const comments = [{ id: 1, text: "Nice post!" }];
    callback(comments);
  }, 1000);
}

// Invocation - callback hell
getUser(1, function (user) {
  console.log("Got user:", user);
  getPosts(user.id, function (posts) {
    console.log("Got posts:", posts);
    getComments(posts[0].id, function (comments) {
      console.log("Got comments:", comments);
    });
  });
});
```

## Real complete implementation same example using promise chain

```js
function getUser(userId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const user = { id: userId, name: "Nishant" };
      resolve(user);
    }, 1000);
  });
}

function getPosts(userId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const posts = [
        { id: 1, title: "Post 1" },
        { id: 2, title: "Post 2" },
      ];
      resolve(posts);
    }, 1000);
  });
}

function getComments(postId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const comments = [{ id: 1, text: "Nice post!" }];
      resolve(comments);
    }, 1000);
  });
}

// Invocation - promise chain
getUser(1)
  .then(function (user) {
    console.log("Got user:", user);
    return getPosts(user.id);
  })
  .then(function (posts) {
    console.log("Got posts:", posts);
    return getComments(posts[0].id);
  })
  .then(function (comments) {
    console.log("Got comments:", comments);
  })
  .catch(function (error) {
    console.log("Error:", error);
  });
```

## Real complete implementation same example using async await

```js
function getUser(userId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const user = { id: userId, name: "Nishant" };
      resolve(user);
    }, 1000);
  });
}

function getPosts(userId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const posts = [
        { id: 1, title: "Post 1" },
        { id: 2, title: "Post 2" },
      ];
      resolve(posts);
    }, 1000);
  });
}

function getComments(postId) {
  return new Promise(function (resolve) {
    setTimeout(function () {
      const comments = [{ id: 1, text: "Nice post!" }];
      resolve(comments);
    }, 1000);
  });
}

// Invocation - async/await
async function getData() {
  const user = await getUser(1);
  console.log("Got user:", user);

  const posts = await getPosts(user.id);
  console.log("Got posts:", posts);

  const comments = await getComments(posts[0].id);
  console.log("Got comments:", comments);
}

getData();
```
