## Core Transformation Terms

### Serialize

**Direction:** Object/Complex â†’ Simple/String  
**Purpose:** Convert complex data to a storable or transmittable format  
**Example:**

```typescript
// Input: Complex object with various formats
{ phone: "(713) 555-1234", email: "John@GMAIL.com" }

// Output: Standardized, API-ready format
{ phone: "+17135551234", email: "john@gmail.com" }
```

**Use Cases:**

- Before sending data to API
- Before storing in database
- Before transmitting over network

---

### Deserialize

**Direction:** Simple/String â†’ Object/Complex  
**Purpose:** Reconstruct structured data from stored format  
**Example:**

```typescript
// Input: E.164 format from database
"+17135551234";

// Output: Form-ready format
"7135551234"; // 10 digits for masked input
```

**Use Cases:**

- Loading data from API into forms
- Reconstructing objects from JSON
- Preparing database values for editing

---

### Parse

**Direction:** String â†’ Structured Data  
**Purpose:** Extract meaning and structure from text  
**Example:**

```typescript

"01/15/2024"
"2024-01-15"
"Jan 15, 2024"

// Output: Structured Date object

// Phone parsing example
"(713) 555-1234" â†’ { areaCode: "713", prefix: "555", line: "1234" }
```

**Use Cases:**

- Parsing dates
- Extracting parts from formatted strings
- Understanding structured text

---

### Stringify

**Direction:** Structured â†’ String  
**Purpose:** Convert structured data to string representation  
**Example:**

```typescript

// Output: JSON string

// Also used for

```

### Format

**Direction:** Data â†’ Human-Readable  
**Purpose:** Make data pretty and readable for display  
**Example:**


### Normalize

## Transformation Terms

### Transform

**Purpose:** General conversion from one form to another  
**Example:**

```typescript

  id: item.id,
  fullName: `${item.firstName} ${item.lastName}`,
```

---

### Convert

**Purpose:** Change type or format  
**Example:**

```typescript
// Type conversion
String(123); // "123"
Number("123"); // 123
Boolean(1); // true

// Unit conversion

// Format conversion

```

---

### Map

**Purpose:** Apply function to each element  
**Example:**

```typescript

// Object mapping

  ...lead,
  phone: formatPhoneForDisplay(lead.phone),

```

---

### Reduce

**Purpose:** Combine many values into one  
**Example:**

```typescript

// Group by

  acc[lead.status] = acc[lead.status] || [];
  acc[lead.status].push(lead);
  return acc;
}, {});
```

---

### Filter

**Purpose:** Remove unwanted items  
**Example:**

```typescript

```

---

### Coerce

**Purpose:** Force type conversion (implicit)  
**Example:**

## Specific Use Cases

### Slugify

**Purpose:** Make URL-friendly strings  
**Example:**

```typescript
function slugify(text: string): string {
  return text
    .toLowerCase()
    .replace(/[^\w\s-]/g, "")
    .replace(/\s+/g, "-")
    .replace(/--+/g, "-")
    .trim();
}

```

### Capitalize

**Purpose:** First letter uppercase  
**Example:**

### Camelize (camelCase)

**Purpose:** Convert to camelCase  
**Example:**

```typescript
function camelize(str: string): string {
  return str
    .replace(/(?:^\w|[A-Z]|\b\w)/g, (letter, index) =>
      index === 0 ? letter.toLowerCase() : letter.toUpperCase()
    )
    .replace(/\s+/g, "");

camelize("first-name"); // "firstName"
```

### Snake_case


```typescript
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// SERIALIZE: User Input â†’ E.164 (API/Database Format)
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const serializePhoneNumber = (
  phone: string | null | undefined
): string | null => {
  if (!phone) return null;

  // Remove all non-digit characters
  const digits = phone.replace(/\D/g, "");

  // Validate: must have 10 digits, or 11 digits starting with 1
  if (digits.length === 10) {
    // US number without country code: add +1
    return `+1${digits}`;
  } else if (digits.length === 11 && digits.startsWith("1")) {
    // US number with country code 1
    return `+${digits}`;
  }

  // Invalid phone number
  return null;
};

// Examples:
serializePhoneNumber("(713) 555-1234"); // â†’ "+17135551234"
serializePhoneNumber("713-555-1234"); // â†’ "+17135551234"
serializePhoneNumber("7135551234"); // â†’ "+17135551234"
serializePhoneNumber("+1 713 555 1234"); // â†’ "+17135551234"

// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// FORMAT: E.164 â†’ Display (Human-Readable)
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const formatPhoneForDisplay = (
  phone: string | null | undefined
): string => {
  if (!phone) return "";

  const digits = phone.replace(/\D/g, "");

  if (digits.length === 10) {
    return `(${digits.slice(0, 3)}) ${digits.slice(3, 6)}-${digits.slice(6)}`;
  } else if (digits.length === 11 && digits.startsWith("1")) {
    const withoutCountryCode = digits.slice(1);
    return `(${withoutCountryCode.slice(0, 3)}) ${withoutCountryCode.slice(
      3,
      6
    )}-${withoutCountryCode.slice(6)}`;
  }

  return phone;
};

// Examples:
formatPhoneForDisplay("+17135551234"); // â†’ "(713) 555-1234"
formatPhoneForDisplay("17135551234"); // â†’ "(713) 555-1234"

// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// DESERIALIZE: E.164 â†’ Form Input (10 digits)
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const deserializePhone = (phone: string | null | undefined): string => {
  if (!phone) return "";

  const digits = phone.replace(/\D/g, "");

  if (digits.length === 11 && digits.startsWith("1")) {
    return digits.slice(1); // Remove country code
  }

  if (digits.length === 10) {
    return digits;
  }

  return "";
};

// Examples:
deserializePhone("+17135551234"); // â†’ "7135551234"
deserializePhone("(713) 555-1234"); // â†’ "7135551234"
```

### Date Transformations

```typescript
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// PARSE: String â†’ ISO 8601
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const parseDateToISO = (
  date: string | Date | null | undefined
): string | null => {
  if (!date) return null;

  try {
    const dateObj = new Date(date);

    if (isNaN(dateObj.getTime())) {
      return null;
    }

    // Set to start of day UTC
    dateObj.setUTCHours(0, 0, 0, 0);

    return dateObj.toISOString();
  } catch {
    return null;
  }
};

// Examples:
parseDateToISO("01/15/2024"); // â†’ "2024-01-15T00:00:00.000Z"
parseDateToISO("2024-01-15"); // â†’ "2024-01-15T00:00:00.000Z"

// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// FORMAT: ISO 8601 â†’ Display
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const formatDateForDisplay = (
  isoDate: string | null | undefined
): string => {
  if (!isoDate) return "";

  try {
    const date = new Date(isoDate);
    return new Intl.DateTimeFormat("en-US", {
      year: "numeric",
      month: "short",
      day: "numeric",
    }).format(date);
  } catch {
    return "";
  }
};

// Examples:
formatDateForDisplay("2024-01-15T00:00:00.000Z"); // â†’ "Jan 15, 2024"
```

### State Normalization

```typescript
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// NORMALIZE: Any Format â†’ Proper Case or Abbreviation
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const normalizeState = (state: string | null | undefined): string => {
  if (!state) return "";
  const trimmed = state.trim();
  if (!trimmed) return "";

  // If 2 characters, assume it's abbreviation and uppercase it
  if (trimmed.length === 2) {
    return trimmed.toUpperCase();
  }

  // Otherwise, return as titlecase
  return trimmed
    .split(" ")
    .map((word) => word.charAt(0).toUpperCase() + word.slice(1).toLowerCase())
    .join(" ");
};

// Examples:
normalizeState("texas"); // â†’ "Texas"
normalizeState("TX"); // â†’ "TX"
normalizeState("CALIFORNIA"); // â†’ "California"
normalizeState("new york"); // â†’ "New York"
```

### Complete Form Serialization

```typescript
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•
// SERIALIZE: Full Contact Data
// â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•

export const serializeContactData = (data: any): any => {
  const normalized: any = { ...data };

  // Normalize phone fields
  if (data.phone) {
    normalized.phone = serializePhoneNumber(data.phone);
  }
  if (data.contactPersonPhone) {
    normalized.contactPersonPhone = serializePhoneNumber(
      data.contactPersonPhone
    );
  }

  // Normalize email
  if (data.email) {
    normalized.email = normalizeEmail(data.email);
  }

  // Normalize ZIP code
  if (data.zip) {
    normalized.zip = normalizeZipCode(data.zip);
  }

  // Normalize state
  if (data.state) {
    normalized.state = normalizeState(data.state);
  }

  // Normalize text fields (trim)
  const textFields = ["fName", "lName", "cName", "street", "city"];
  textFields.forEach((field) => {
    if (data[field] !== undefined) {
      normalized[field] = data[field]?.trim() || "";
    }
  });

  // Normalize dates
  if (data.deliveryDate) {
    normalized.deliveryDate = parseDateToISO(data.deliveryDate);
  }
  if (data.pickupDate) {
    normalized.pickupDate = parseDateToISO(data.pickupDate);
  }

  return normalized;
};
```
